// =============================== // LAMBO CITY — SYSTEM CORE (PHASE 7.5) // =============================== // 🎮 CORE STATE const GameState = { currentZone: "cabin", playerState: "idle", isTransitioning: false }; // 🧍 PLAYER SYSTEM const Player = { velocity: { x: 0, z: 0 }, speed: 0.1, position: null }; // 🌍 WORLD SYSTEM const World = { yachtFocus: null, dockActive: false }; // 🎮 INPUT SYSTEM const Input = { keys: {} }; document.addEventListener("keydown", e => { Input.keys[e.key.toLowerCase()] = true; }); document.addEventListener("keyup", e => { Input.keys[e.key.toLowerCase()] = false; }); // =============================== // PLAYER UPDATE SYSTEM // =============================== function updatePlayer() { let targetX = 0; let targetZ = 0; if (Input.keys["w"]) targetZ -= Player.speed; if (Input.keys["s"]) targetZ += Player.speed; if (Input.keys["a"]) targetX -= Player.speed; if (Input.keys["d"]) targetX += Player.speed; Player.velocity.x += (targetX - Player.velocity.x) * 0.2; Player.velocity.z += (targetZ - Player.velocity.z) * 0.2; player.position.x += Player.velocity.x; player.position.z += Player.velocity.z; Player.position = player.position; } // =============================== // CAMERA SYSTEM // =============================== function updateCamera() { camera.position.x += (player.position.x - camera.position.x) * 0.05; camera.position.z += (player.position.z + 5 - camera.position.z) * 0.05; camera.lookAt(player.position); } // =============================== // GAME STATE SYSTEM // =============================== function updateGameState() { let dist = Math.hypot(player.position.x, player.position.z + 6); if (GameState.currentZone === "cabin" && dist < 2.5) { document.getElementById("prompt").style.display = "block"; } else { document.getElementById("prompt").style.display = "none"; } document.getElementById("zone").innerText = "ZONE: " + GameState.currentZone.toUpperCase(); document.getElementById("state").innerText = "STATE: " + GameState.playerState.toUpperCase(); } // =============================== // TRANSITION SYSTEM // =============================== function enterDock() { if (GameState.isTransitioning) return; GameState.isTransitioning = true; document.getElementById("fade").style.opacity = 1; setTimeout(() => { GameState.currentZone = "dock"; document.getElementById("fade").style.opacity = 0; GameState.isTransitioning = false; }, 1000); } // =============================== // MAIN LOOP CLEAN STRUCTURE // =============================== function gameLoop() { requestAnimationFrame(gameLoop); updatePlayer(); updateCamera(); updateGameState(); renderer.render(scene, camera); }
